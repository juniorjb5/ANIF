def scenario_gain_loss(portfolio_value, portfolio_std_dev, z_score, portfolio_expected_return, days):
    """
    Calcula la ganancia o pÃ©rdida del portafolio en un escenario simulado.

    ParÃ¡metros:
    - portfolio_value: Valor total del portafolio.
    - portfolio_std_dev: DesviaciÃ³n estÃ¡ndar (volatilidad) del portafolio.
